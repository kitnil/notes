* Convert MariaDB to JSON

** Select database
#+begin_src sql
  use powerdns;
#+end_src

** Depending on MariaDB version
#+begin_src sql
  SELECT
    JSON_OBJECT(
      'id', id,
      'domain_id', domain_id,
      'name', name,
      'type', type,
      'content', content,
      'ttl', ttl,
      'priority', prio,
      'auth', auth
    ) AS json_data
  FROM records;
#+end_src

#+begin_src sql
  SELECT CONCAT(
    '{',
    '"id":', id, ',',
    '"domain_id":', domain_id, ',',
    '"name":"', name, '",',
    '"type":"', type, '",',
    '"content":"', content, '",',
    '"ttl":', ttl, ',',
    '"priority":', prio, ',',
    '"auth":', auth,
    '}'
  ) AS json_data
  FROM records;
#+end_src
