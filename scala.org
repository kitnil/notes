** Write me a Scala program with a shebung line to get https://www.gnu.org/ HTML page.

import scala.io._
import java.net.URL

object GetHTML {

  def main(args: Array[String]): Unit = {

    val url = new URL("https://www.gnu.org/")
    val htmlString = Source.fromURL(url).mkString
    println(htmlString)
  }

}
